/*
 * Jeep LEDs Controller - NO STRING VERSION
 *
 * This version completely avoids the Arduino String class
 * which can be unreliable on Arduino Uno with limited memory.
 *
 * Uses only char arrays and direct parsing.
 */

#include <Adafruit_NeoPixel.h>
#include <SoftwareSerial.h>

// Configuration
#define LED_PIN        6
#define NUM_LEDS       60
#define BLUEFRUIT_RX   10
#define BLUEFRUIT_TX   11
#define CMD_BUFFER_SIZE 32

// Initialize Neopixel strip
Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

// Initialize Software Serial for Bluefruit
SoftwareSerial bluefruitSerial = SoftwareSerial(BLUEFRUIT_RX, BLUEFRUIT_TX);

// State variables
uint8_t red = 0, green = 0, blue = 0;
uint8_t brightness = 255;
unsigned long lastUpdate = 0;

// Command buffer
char cmdBuffer[CMD_BUFFER_SIZE];
uint8_t cmdBufferIndex = 0;
unsigned long lastCharTime = 0;
unsigned long commandCount = 0;

void setup() {
  Serial.begin(9600);
  delay(1000);

  Serial.println("==================================");
  Serial.println("Jeep LEDs - NO STRING VERSION");
  Serial.println("==================================");

  bluefruitSerial.begin(9600);
  Serial.println("✓ Bluefruit: 9600 baud");

  strip.begin();
  strip.setBrightness(brightness);
  strip.show();

  Serial.print("✓ Neopixel: ");
  Serial.print(NUM_LEDS);
  Serial.println(" LEDs");
  Serial.println("==================================");
  Serial.println("READY");
  Serial.println("==================================\n");
}

void loop() {
  readBluetoothCommand();
}

void readBluetoothCommand() {
  while (bluefruitSerial.available()) {
    char c = bluefruitSerial.read();
    lastCharTime = millis();

    Serial.print("RX: '");
    Serial.print(c);
    Serial.print("' (0x");
    Serial.print((int)c, HEX);
    Serial.println(")");

    // End of command
    if (c == '\n' || c == '\r') {
      if (cmdBufferIndex > 0) {
        cmdBuffer[cmdBufferIndex] = '\0';

        Serial.println("\n━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
        Serial.print("📥 COMMAND #");
        Serial.println(++commandCount);
        Serial.print("   Buffer: '");
        Serial.print(cmdBuffer);
        Serial.print("' (");
        Serial.print(cmdBufferIndex);
        Serial.println(" bytes)");

        // Process command directly from char array
        processCommandDirect(cmdBuffer, cmdBufferIndex);

        Serial.println("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n");

        cmdBufferIndex = 0;
      }
    }
    // Add to buffer
    else if (cmdBufferIndex < CMD_BUFFER_SIZE - 1) {
      cmdBuffer[cmdBufferIndex++] = c;
    }
    else {
      Serial.println("❌ Buffer overflow!");
      cmdBufferIndex = 0;
    }
  }

  // Timeout
  if (cmdBufferIndex > 0 && (millis() - lastCharTime) > 100) {
    Serial.println("⚠️  Timeout - resetting buffer");
    cmdBufferIndex = 0;
  }
}

void processCommandDirect(char* cmd, int len) {
  if (len == 0) {
    Serial.println("⚠️  Empty command");
    return;
  }

  char cmdType = cmd[0];
  Serial.print("🔍 Type: '");
  Serial.print(cmdType);
  Serial.println("'");

  switch (cmdType) {
    case 'C': // Color command
      Serial.println("   → COLOR");
      parseColorDirect(cmd + 1, len - 1);  // Skip 'C', pass rest

      Serial.print("✅ RGB: (");
      Serial.print(red);
      Serial.print(",");
      Serial.print(green);
      Serial.print(",");
      Serial.print(blue);
      Serial.println(")");

      // Update LEDs immediately
      Serial.println("   → Updating LEDs NOW");
      updateLEDs();
      Serial.println("   → LEDs updated!");
      break;

    case 'B': // Brightness
      Serial.println("   → BRIGHTNESS");
      brightness = parseIntDirect(cmd + 1, len - 1);
      brightness = constrain(brightness, 0, 255);
      strip.setBrightness(brightness);
      strip.show();
      Serial.print("✅ Brightness: ");
      Serial.println(brightness);
      break;

    case 'P': // Pattern
      Serial.println("   → PATTERN (ignored for now)");
      break;

    case 'S': // Speed
      Serial.println("   → SPEED (ignored for now)");
      break;

    default:
      Serial.print("❌ Unknown: '");
      Serial.print(cmdType);
      Serial.println("'");
      break;
  }
}

void parseColorDirect(char* data, int len) {
  // Parse "255,127,0" directly from char array
  int values[3] = {0, 0, 0};
  int valueIndex = 0;
  int currentValue = 0;

  for (int i = 0; i < len && valueIndex < 3; i++) {
    char c = data[i];

    if (c >= '0' && c <= '9') {
      currentValue = currentValue * 10 + (c - '0');
    }
    else if (c == ',') {
      values[valueIndex++] = currentValue;
      currentValue = 0;
    }
  }

  // Last value (no comma after it)
  if (valueIndex < 3) {
    values[valueIndex] = currentValue;
  }

  red = constrain(values[0], 0, 255);
  green = constrain(values[1], 0, 255);
  blue = constrain(values[2], 0, 255);

  Serial.print("   Parsed: ");
  Serial.print(values[0]);
  Serial.print(",");
  Serial.print(values[1]);
  Serial.print(",");
  Serial.println(values[2]);
}

int parseIntDirect(char* data, int len) {
  int result = 0;
  for (int i = 0; i < len; i++) {
    char c = data[i];
    if (c >= '0' && c <= '9') {
      result = result * 10 + (c - '0');
    }
  }
  return result;
}

void updateLEDs() {
  Serial.print("   Setting all ");
  Serial.print(NUM_LEDS);
  Serial.print(" LEDs to ");
  Serial.print(red);
  Serial.print(",");
  Serial.print(green);
  Serial.print(",");
  Serial.println(blue);

  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, strip.Color(red, green, blue));
  }

  Serial.println("   Calling strip.show()...");
  strip.show();
  Serial.println("   strip.show() complete");
}
