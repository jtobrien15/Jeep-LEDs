/*
 * Neopixel Hardware Test
 *
 * This sketch tests your Neopixel strip hardware without Bluetooth.
 * It will cycle through colors to verify your wiring and configuration.
 *
 * If this doesn't work, you have a hardware issue with the Neopixels.
 */

#include <Adafruit_NeoPixel.h>

// CONFIGURATION - Try changing these if LEDs don't work
#define LED_PIN        6        // Try pins: 6, 5, 9, 10, 11
#define NUM_LEDS       30       // Change to actual number of LEDs

// Try different LED types if colors are wrong:
// NEO_GRB + NEO_KHZ800  (most common)
// NEO_RGB + NEO_KHZ800
// NEO_GRBW + NEO_KHZ800
Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

void setup() {
  Serial.begin(9600);
  delay(1000);

  Serial.println("========================================");
  Serial.println("Neopixel Hardware Test");
  Serial.println("========================================");
  Serial.print("Pin: ");
  Serial.println(LED_PIN);
  Serial.print("Number of LEDs: ");
  Serial.println(NUM_LEDS);
  Serial.println("========================================");

  strip.begin();
  strip.setBrightness(50);  // Start dim for safety
  strip.show(); // Initialize all LEDs to 'off'

  Serial.println("Starting LED test...");
  Serial.println("Watch your LED strip!");
}

void loop() {
  // Test 1: Turn all LEDs RED
  Serial.println("\n[Test 1] All LEDs RED");
  setAllColor(255, 0, 0);
  delay(2000);

  // Test 2: Turn all LEDs GREEN
  Serial.println("[Test 2] All LEDs GREEN");
  setAllColor(0, 255, 0);
  delay(2000);

  // Test 3: Turn all LEDs BLUE
  Serial.println("[Test 3] All LEDs BLUE");
  setAllColor(0, 0, 255);
  delay(2000);

  // Test 4: Turn all LEDs WHITE
  Serial.println("[Test 4] All LEDs WHITE");
  setAllColor(255, 255, 255);
  delay(2000);

  // Test 5: Turn all LEDs OFF
  Serial.println("[Test 5] All LEDs OFF");
  setAllColor(0, 0, 0);
  delay(2000);

  // Test 6: Light up LEDs one by one
  Serial.println("[Test 6] Lighting LEDs one by one (RED)");
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, strip.Color(255, 0, 0));
    strip.show();
    Serial.print("LED ");
    Serial.print(i);
    Serial.println(" ON");
    delay(100);
  }
  delay(1000);

  // Test 7: Turn off LEDs one by one
  Serial.println("[Test 7] Turning off LEDs one by one");
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, strip.Color(0, 0, 0));
    strip.show();
    Serial.print("LED ");
    Serial.print(i);
    Serial.println(" OFF");
    delay(100);
  }
  delay(1000);

  Serial.println("\n========================================");
  Serial.println("Test cycle complete. Repeating...");
  Serial.println("========================================");
}

void setAllColor(uint8_t r, uint8_t g, uint8_t b) {
  Serial.print("  Setting RGB(");
  Serial.print(r);
  Serial.print(", ");
  Serial.print(g);
  Serial.print(", ");
  Serial.print(b);
  Serial.println(")");

  for (int i = 0; i < strip.numPixels(); i++) {
    strip.setPixelColor(i, strip.Color(r, g, b));
  }
  strip.show();

  Serial.println("  strip.show() called");
}
