/*
 * Jeep LEDs Controller - Final Production Version
 *
 * This version uses direct char array parsing instead of the Arduino String class
 * for maximum reliability on Arduino Uno with limited RAM.
 *
 * Hardware Requirements:
 * - Arduino Uno R2
 * - Adafruit Bluefruit LE UART Friend
 * - Adafruit Neopixel LED strip
 *
 * Wiring:
 * - Bluefruit TX -> Arduino pin 10 (BLUEFRUIT_RX)
 * - Bluefruit RX -> Arduino pin 11 (BLUEFRUIT_TX)
 * - Neopixel Data -> Arduino pin 6
 * - Neopixel Power -> External 5V power supply (share ground with Arduino)
 *
 * Libraries needed (install via Arduino Library Manager):
 * - Adafruit NeoPixel
 *
 * Command Protocol:
 * - C<red>,<green>,<blue>\n - Set solid color (e.g., "C255,0,0\n" for red)
 * - P<pattern>\n - Set pattern (e.g., "PRAINBOW\n")
 * - B<brightness>\n - Set brightness 0-255 (e.g., "B128\n")
 * - S<speed>\n - Set speed percentage (e.g., "S100\n")
 */

#include <Adafruit_NeoPixel.h>
#include <SoftwareSerial.h>

// Configuration
#define LED_PIN        6        // Pin connected to Neopixel data line
#define NUM_LEDS       60       // Number of LEDs in your strip
#define BLUEFRUIT_RX   10       // Bluefruit UART RX (connect to Bluefruit TX)
#define BLUEFRUIT_TX   11       // Bluefruit UART TX (connect to Bluefruit RX)
#define CMD_BUFFER_SIZE 32      // Maximum command length

// Initialize Neopixel strip
Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUM_LEDS, LED_PIN, NEO_GRB + NEO_KHZ800);

// Initialize Software Serial for Bluefruit
SoftwareSerial bluefruitSerial = SoftwareSerial(BLUEFRUIT_RX, BLUEFRUIT_TX);

// State variables
uint8_t red = 0, green = 0, blue = 0;  // Current color
uint8_t brightness = 255;
char currentPattern[16] = "SOLID";     // Current pattern name
unsigned long lastUpdate = 0;
int patternStep = 0;
float speedMultiplier = 1.0;  // 1.0 = normal, 0.5 = fast, 2.0 = slow

// Command buffer for non-blocking reads
char cmdBuffer[CMD_BUFFER_SIZE];
uint8_t cmdBufferIndex = 0;
unsigned long lastCharTime = 0;

void setup() {
  // Initialize Serial for debugging
  Serial.begin(9600);
  delay(1000);

  Serial.println("==================================");
  Serial.println("Jeep LEDs Controller v3.0");
  Serial.println("==================================");

  // Initialize Bluefruit UART
  bluefruitSerial.begin(9600);
  Serial.println("Bluefruit Serial: 9600 baud");

  // Initialize Neopixel strip
  strip.begin();
  strip.setBrightness(brightness);
  strip.show();

  Serial.print("Neopixel: ");
  Serial.print(NUM_LEDS);
  Serial.println(" LEDs ready");

  Serial.println("==================================");
  Serial.println("Ready to receive commands");
  Serial.println("==================================\n");
}

void loop() {
  // Non-blocking command reading
  readBluetoothCommand();

  // Update LED pattern
  updatePattern();
}

void readBluetoothCommand() {
  // Read one character at a time without blocking
  while (bluefruitSerial.available()) {
    char c = bluefruitSerial.read();
    lastCharTime = millis();

    // End of command
    if (c == '\n' || c == '\r') {
      if (cmdBufferIndex > 0) {
        cmdBuffer[cmdBufferIndex] = '\0';  // Null terminate

        Serial.print("Received: '");
        Serial.print(cmdBuffer);
        Serial.println("'");

        // Process command directly from char array (no String class)
        processCommand(cmdBuffer, cmdBufferIndex);

        cmdBufferIndex = 0;  // Reset buffer
      }
    }
    // Add to buffer if not full
    else if (cmdBufferIndex < CMD_BUFFER_SIZE - 1) {
      cmdBuffer[cmdBufferIndex++] = c;
    }
    // Buffer overflow - reset
    else {
      Serial.println("ERROR: Command buffer overflow");
      cmdBufferIndex = 0;
    }
  }

  // Timeout - if we've been building a command for >100ms without newline, reset
  if (cmdBufferIndex > 0 && (millis() - lastCharTime) > 100) {
    Serial.println("WARNING: Command timeout, buffer reset");
    cmdBufferIndex = 0;
  }
}

void processCommand(char* cmd, int len) {
  if (len == 0) return;

  char cmdType = cmd[0];

  switch (cmdType) {
    case 'C': // Color command
      parseColor(cmd + 1, len - 1);
      Serial.print("Color set to R:");
      Serial.print(red);
      Serial.print(" G:");
      Serial.print(green);
      Serial.print(" B:");
      Serial.println(blue);
      break;

    case 'P': // Pattern command
      copyString(currentPattern, cmd + 1, len - 1, 15);
      patternStep = 0;
      Serial.print("Pattern set to: ");
      Serial.println(currentPattern);
      break;

    case 'B': // Brightness command
      brightness = parseInt(cmd + 1, len - 1);
      brightness = constrain(brightness, 0, 255);
      strip.setBrightness(brightness);
      strip.show();
      Serial.print("Brightness set to: ");
      Serial.println(brightness);
      break;

    case 'S': // Speed command (percentage: 50=fast, 100=normal, 200=slow)
      int speedPercent = parseInt(cmd + 1, len - 1);
      speedMultiplier = speedPercent / 100.0;
      speedMultiplier = constrain(speedMultiplier, 0.25, 4.0);
      Serial.print("Speed set to: ");
      Serial.print(speedPercent);
      Serial.println("%");
      break;

    default:
      Serial.print("ERROR: Unknown command type: ");
      Serial.println(cmdType);
      break;
  }
}

void parseColor(char* data, int len) {
  // Parse "255,127,0" directly from char array
  int values[3] = {0, 0, 0};
  int valueIndex = 0;
  int currentValue = 0;

  for (int i = 0; i < len && valueIndex < 3; i++) {
    char c = data[i];

    if (c >= '0' && c <= '9') {
      currentValue = currentValue * 10 + (c - '0');
    }
    else if (c == ',') {
      values[valueIndex++] = currentValue;
      currentValue = 0;
    }
  }

  // Last value (no comma after it)
  if (valueIndex < 3) {
    values[valueIndex] = currentValue;
  }

  red = constrain(values[0], 0, 255);
  green = constrain(values[1], 0, 255);
  blue = constrain(values[2], 0, 255);
}

int parseInt(char* data, int len) {
  int result = 0;
  for (int i = 0; i < len; i++) {
    char c = data[i];
    if (c >= '0' && c <= '9') {
      result = result * 10 + (c - '0');
    }
  }
  return result;
}

void copyString(char* dest, char* src, int srcLen, int maxLen) {
  int copyLen = min(srcLen, maxLen);
  for (int i = 0; i < copyLen; i++) {
    dest[i] = src[i];
  }
  dest[copyLen] = '\0';
}

bool compareString(char* str1, const char* str2) {
  int i = 0;
  while (str1[i] != '\0' && str2[i] != '\0') {
    if (str1[i] != str2[i]) return false;
    i++;
  }
  return str1[i] == str2[i];  // Both must end at same time
}

void updatePattern() {
  unsigned long currentMillis = millis();

  if (compareString(currentPattern, "SOLID")) {
    // Solid color - only update when color changes
    static uint8_t lastR = 255, lastG = 255, lastB = 255;
    if (red != lastR || green != lastG || blue != lastB) {
      solidColor(red, green, blue);
      lastR = red;
      lastG = green;
      lastB = blue;
    }
  }
  else if (compareString(currentPattern, "BLINK")) {
    if (currentMillis - lastUpdate >= (500 * speedMultiplier)) {
      lastUpdate = currentMillis;
      patternStep = !patternStep;
      solidColor(patternStep ? red : 0, patternStep ? green : 0, patternStep ? blue : 0);
    }
  }
  else if (compareString(currentPattern, "BREATHE")) {
    if (currentMillis - lastUpdate >= (20 * speedMultiplier)) {
      lastUpdate = currentMillis;
      float breath = (exp(sin(millis() / (2000.0 * speedMultiplier) * PI)) - 0.36787944) * 108.0;
      uint8_t val = map(breath, 0, 255, 0, 255);
      solidColor(red * val / 255, green * val / 255, blue * val / 255);
    }
  }
  else if (compareString(currentPattern, "STROBE")) {
    if (currentMillis - lastUpdate >= (100 * speedMultiplier)) {
      lastUpdate = currentMillis;
      patternStep = !patternStep;
      solidColor(patternStep ? red : 0, patternStep ? green : 0, patternStep ? blue : 0);
    }
  }
  else if (compareString(currentPattern, "POLICE")) {
    if (currentMillis - lastUpdate >= (100 * speedMultiplier)) {
      lastUpdate = currentMillis;
      patternStep = !patternStep;
      policeFlash(patternStep);
    }
  }
  else if (compareString(currentPattern, "HAZARD")) {
    if (currentMillis - lastUpdate >= (400 * speedMultiplier)) {
      lastUpdate = currentMillis;
      patternStep = !patternStep;
      solidColor(patternStep ? 255 : 0, patternStep ? 191 : 0, 0);
    }
  }
  else if (compareString(currentPattern, "ALTERNATE")) {
    if (currentMillis - lastUpdate >= (300 * speedMultiplier)) {
      lastUpdate = currentMillis;
      patternStep = !patternStep;
      alternateFlash(patternStep);
    }
  }
  else if (compareString(currentPattern, "RAINBOW")) {
    if (currentMillis - lastUpdate >= (20 * speedMultiplier)) {
      lastUpdate = currentMillis;
      rainbow(patternStep);
      patternStep = (patternStep + 1) % 256;
    }
  }
  else if (compareString(currentPattern, "CHASE")) {
    if (currentMillis - lastUpdate >= (50 * speedMultiplier)) {
      lastUpdate = currentMillis;
      chase(red, green, blue, patternStep);
      patternStep = (patternStep + 1) % NUM_LEDS;
    }
  }
  else if (compareString(currentPattern, "SPARKLE")) {
    if (currentMillis - lastUpdate >= (100 * speedMultiplier)) {
      lastUpdate = currentMillis;
      sparkle(red, green, blue);
    }
  }
  else if (compareString(currentPattern, "FADE")) {
    if (currentMillis - lastUpdate >= (30 * speedMultiplier)) {
      lastUpdate = currentMillis;
      uint8_t val = (sin(millis() / (1000.0 * speedMultiplier)) + 1.0) * 127.5;
      solidColor(red * val / 255, green * val / 255, blue * val / 255);
    }
  }
  else if (compareString(currentPattern, "RUNNING")) {
    if (currentMillis - lastUpdate >= (100 * speedMultiplier)) {
      lastUpdate = currentMillis;
      runningLights(red, green, blue, patternStep);
      patternStep = (patternStep + 1) % 3;
    }
  }
}

// Pattern functions
void solidColor(uint8_t r, uint8_t g, uint8_t b) {
  for (int i = 0; i < NUM_LEDS; i++) {
    strip.setPixelColor(i, strip.Color(r, g, b));
  }
  strip.show();
}

void rainbow(uint8_t offset) {
  for (int i = 0; i < strip.numPixels(); i++) {
    strip.setPixelColor(i, Wheel((i + offset) & 255));
  }
  strip.show();
}

void chase(uint8_t r, uint8_t g, uint8_t b, int position) {
  for (int i = 0; i < strip.numPixels(); i++) {
    if (i == position) {
      strip.setPixelColor(i, strip.Color(r, g, b));
    } else {
      strip.setPixelColor(i, strip.Color(0, 0, 0));
    }
  }
  strip.show();
}

void sparkle(uint8_t r, uint8_t g, uint8_t b) {
  int pixel = random(strip.numPixels());
  strip.setPixelColor(pixel, strip.Color(r, g, b));
  strip.show();
  delay(50);
  strip.setPixelColor(pixel, strip.Color(0, 0, 0));
}

void policeFlash(bool step) {
  int halfPoint = NUM_LEDS / 2;
  for (int i = 0; i < NUM_LEDS; i++) {
    if (i < halfPoint) {
      strip.setPixelColor(i, step ? strip.Color(255, 0, 0) : strip.Color(0, 0, 0));
    } else {
      strip.setPixelColor(i, step ? strip.Color(0, 0, 0) : strip.Color(0, 0, 255));
    }
  }
  strip.show();
}

void alternateFlash(bool step) {
  int halfPoint = NUM_LEDS / 2;
  for (int i = 0; i < NUM_LEDS; i++) {
    if (i < halfPoint) {
      strip.setPixelColor(i, step ? strip.Color(red, green, blue) : strip.Color(0, 0, 0));
    } else {
      strip.setPixelColor(i, step ? strip.Color(0, 0, 0) : strip.Color(red, green, blue));
    }
  }
  strip.show();
}

void runningLights(uint8_t r, uint8_t g, uint8_t b, int step) {
  for (int i = 0; i < NUM_LEDS; i++) {
    if ((i % 3) == step) {
      strip.setPixelColor(i, strip.Color(r, g, b));
    } else {
      strip.setPixelColor(i, strip.Color(0, 0, 0));
    }
  }
  strip.show();
}

uint32_t Wheel(byte WheelPos) {
  WheelPos = 255 - WheelPos;
  if (WheelPos < 85) {
    return strip.Color(255 - WheelPos * 3, 0, WheelPos * 3);
  }
  if (WheelPos < 170) {
    WheelPos -= 85;
    return strip.Color(0, WheelPos * 3, 255 - WheelPos * 3);
  }
  WheelPos -= 170;
  return strip.Color(WheelPos * 3, 255 - WheelPos * 3, 0);
}
